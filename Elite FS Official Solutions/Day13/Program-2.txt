import java.util.*;

class BCS {
	public List<Integer> binaryCode(int n) {
		List<Integer> rs=new ArrayList<Integer>();
		rs.add(0);
		for(int i=0;i<n;i++){
			int size=rs.size();
			for(int k=size-1;k>=0;k--)
				rs.add(rs.get(k) | 1<<i);
		}
		return rs;
	}
	public static void main( String args[])
	{
		Scanner sc=new Scanner(System.in);
		int N=sc.nextInt();
		System.out.println(new BCS().binaryCode(N));
	}
}
